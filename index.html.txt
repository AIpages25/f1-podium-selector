    <script>
        const drivers2025 = [
            { name: "Max Verstappen", team: "Red Bull Racing" },
            { name: "Yuki Tsunoda", team: "Red Bull Racing" },
            { name: "Charles Leclerc", team: "Ferrari" },
            { name: "Lewis Hamilton", team: "Ferrari" },
            { name: "George Russell", team: "Mercedes" },
            { name: "Andrea Kimi Antonelli", team: "Mercedes" },
            { name: "Lando Norris", team: "McLaren" },
            { name: "Oscar Piastri", team: "McLaren" },
            { name: "Fernando Alonso", team: "Aston Martin Aramco F1 Team" },
            { name: "Lance Stroll", team: "Aston Martin Aramco F1 Team" },
            { name: "Pierre Gasly", team: "Alpine F1 Team" },
            { name: "Franco Colapinto", team: "Alpine F1 Team" },
            { name: "Alexander Albon", team: "Williams Racing" },
            { name: "Carlos Sainz", team: "Williams Racing" },
            { name: "Liam Lawson", team: "RB (Racing Bulls)" },
            { name: "Isack Hadjar", team: "RB (Racing Bulls)" },
            { name: "Nico Hülkenberg", team: "Sauber" },
            { name: "Gabriel Bortoleto", team: "Sauber" },
            { name: "Esteban Ocon", team: "Haas F1 Team" },
            { name: "Oliver Bearman", team: "Haas F1 Team" }
        ];

        const firstPlaceSelect = document.getElementById('firstPlace');
        const secondPlaceSelect = document.getElementById('secondPlace');
        const thirdPlaceSelect = document.getElementById('thirdPlace');
        const form = document.getElementById('podiumPredictorForm');
        const predictionOutput = document.getElementById('predictionOutput');

        // Function to populate select dropdowns
        function populateDropdowns(selectElement, excludeDrivers = []) {
            selectElement.innerHTML = '<option value="">-- Select Driver --</option>'; // Reset
            const currentSelectedValue = selectElement.value; // Store current value to reselect later

            drivers2025.forEach(driver => {
                // Only add driver if not in the exclude list
                if (!excludeDrivers.includes(driver.name)) {
                    const option = document.createElement('option');
                    option.value = driver.name;
                    option.textContent = `${driver.name} (${driver.team})`;
                    option.classList.add('driver-option');
                    selectElement.appendChild(option);
                }
            });
            // Attempt to reselect the previously selected value if it's still available
            if (currentSelectedValue && !excludeDrivers.includes(currentSelectedValue)) {
                selectElement.value = currentSelectedValue;
            } else {
                selectElement.value = ""; // Reset if selected driver is now excluded
            }
        }

        // Initial population of all dropdowns
        populateDropdowns(firstPlaceSelect);
        populateDropdowns(secondPlaceSelect);
        populateDropdowns(thirdPlaceSelect);

        // Event listener for changes in dropdowns to prevent duplicate selections
        form.addEventListener('change', (event) => {
            const first = firstPlaceSelect.value;
            const second = secondPlaceSelect.value;
            const third = thirdPlaceSelect.value;

            // Re-populate dropdowns excluding selected drivers
            populateDropdowns(firstPlaceSelect, [second, third].filter(Boolean));
            populateDropdowns(secondPlaceSelect, [first, third].filter(Boolean));
            populateDropdowns(thirdPlaceSelect, [first, second].filter(Boolean));
        });

        // Handle form submission
        form.addEventListener('submit', (event) => {
            event.preventDefault(); // Prevent default form submission

            const first = firstPlaceSelect.value;
            const second = secondPlaceSelect.value;
            const third = thirdPlaceSelect.value;

            // Basic validation for unique selections
            if (first === second || first === third || second === third) {
                alert("Please select three different drivers for the podium positions.");
                return;
            }

            if (first && second && third) {
                predictionOutput.innerHTML = `
                    <p><strong>1st Place:</strong> ${first}</p>
                    <p><strong>2nd Place:</strong> ${second}</p>
                    <p><strong>3rd Place:</strong> ${third}</p>
                `;
            } else {
                predictionOutput.textContent = "Please select drivers for all three podium positions.";
            }
        });
    </script>
